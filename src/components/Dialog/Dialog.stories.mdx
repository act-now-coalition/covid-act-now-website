import { Meta, Story } from '@storybook/addon-docs/blocks';
import Dialog, { useDialog } from './';
import { DialogProps } from './Dialog';

<Meta title="Building Blocks/Dialog" />

export const renderTitle = () => <h1>Title</h1>;

export const FramedDialog = props => {
  const [isOpen, openDialog, closeDialog] = useDialog(false);
  return (
    <div>
      <button onClick={openDialog}>open</button>
      <button onClick={closeDialog}>close</button>
      <Dialog open={isOpen} closeDialog={closeDialog} {...props} />
    </div>
  );
};

<Story name="Open">
  <FramedDialog renderHeader={renderTitle} fullWidth maxWidth="md">
    Our goal is to provide you with clear, trusted information so that you can
    make informed decisions and help stop the pandemic.
  </FramedDialog>
</Story>
